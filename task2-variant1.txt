import React, { useState } from 'react';

const Counter = () => {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Значение: {count}</p>
      <button
        onClick={() => setCount(count + 1)}
        style={{ 
          backgroundColor: count > 10 ? 'red' : 'green',
          color: 'white'
        }}
      >
        +1
      </button>
    </div>
  );
};

export default Counter;
